# @task: Implementar pipeline extracción frames MP4(H.264, 30fps)
# @constraints: OpenCV, memoria máxima 512MB, procesamiento por chunks
# @output: Secuencia PNG numerada sin pérdida

import cv2
import numpy as np

# Estrategia: Lectura por bloques para videos >4K sin OOM
def extract_frames_streaming(input_path, output_pattern, fps=1):
    cap = cv2.VideoCapture(input_path)
    frame_count = 0
    while cap.isOpened():
        ret, frame = cap.read()
        if not ret: break
        if frame_count % (30 // fps) == 0:
            cv2.imwrite(output_pattern.format(frame_count), frame)
        frame_count += 1
    cap.release()

Zero Bloat JavaScript — React Hook

// @instrucción: custom hook useDebounce con TypeScript.
// Incluye cancelación. Sin comentarios extra. T=0.1

import { useState, useEffect, useRef } from 'react';

export function useDebounce<T>(value: T, delay: number): T {
  const [debouncedValue, setDebouncedValue] = useState<T>(value);
  const timerRef = useRef<ReturnType<typeof setTimeout>>();

  useEffect(() => {
    timerRef.current = setTimeout(() => setDebouncedValue(value), delay);
    return () => clearTimeout(timerRef.current);
  }, [value, delay]);

  return debouncedValue;
}

SQL Quirúrgico — One-liner

-- TOP 5 clientes 2024. Tablas: clientes, pedidos.
-- Incluye CTE. Solo query. Sin explicación.

WITH cliente_gasto AS (
  SELECT c.id, c.nombre, SUM(p.total) AS total_gastado
  FROM clientes c
  JOIN pedidos p ON c.id = p.cliente_id
  WHERE p.fecha BETWEEN '2024-01-01' AND '2024-12-31'
  GROUP BY c.id, c.nombre
)
SELECT * FROM cliente_gasto
ORDER BY total_gastado DESC
LIMIT 5;

Python — Función Optimizada

# Escribe función fibonacci(n) recursiva con memoización
# Docstring. Type hints. Sin explicación.

def fibonacci(n: int, memo: dict = None) -> int:
    """Retorna el n-ésimo número de Fibonacci."""
    if memo is None: memo = {}
    if n in memo: return memo[n]
    if n <= 2: return 1
    memo[n] = fibonacci(n-1, memo) + fibonacci(n-2, memo)
    return memo[n]

Decir "actúa como un experto" sin definir qué expertise, en qué sector, con qué metodología = rol vacío que no filtra nada.

          

            
            ~40 tokens/iteración desperdiciados
          

        


        

          

            ❌
            
Pedir "hazlo mejor" sin especificar

          

          
La IA no sabe qué es "mejor" para ti. Di exactamente: "reduce longitud 50%", "cambia tono a formal".

          

            
            ~150 tokens/ciclo desperdiciados
          

        


        

          

            ❌
            
No limitar longitud de respuesta

          

          
Sin límite, la IA genera hasta el máximo permitido. Si necesitas 3 líneas y genera 20 párrafos = tokens quemados.

          

            
            ~800 tokens/exceso desperdiciados
          

        


        

          

            ⚠️
            
Pegar contexto sin delimitar

          

          
Si pegas 2000 palabras de contexto sueltas, la IA no sabe dónde empieza tu instrucción. Usa siempre delimitadores.

          

            
            ~300 tokens de confusión
          

        


        

          

            ⚠️
            
Usar "piensa paso a paso" sin necesidad

          

          
CoT quema 200-500 tokens extra. Úsalo SOLO para matemáticas o lógica compleja. Para texto = desperdicio.

          

            
            ~350 tokens innecesarios
          

        


        

          

            ⚠️
            
Repetir contexto en cada prompt

          

          
No repitas en cada prompt el contexto completo del proyecto. Usa un "system prompt" o bloque de contexto fijo.

          

            
            ~200 tokens/prompt desperdiciados
